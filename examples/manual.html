const fireworks = new GrandFireworks({
  performance: { preset: 'high', adaptive: true },
  finale: { enabled: true }
});

fireworks.start();
fireworks.launch();
fireworks.launchText('Grand finale!');
fireworks.launchFinale({ stopAfter: false });
await fireworks.stop();
